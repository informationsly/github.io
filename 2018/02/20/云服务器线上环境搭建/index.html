sudo setsebool -P ftp_home_dir 1

安装nginx

cd /developer/setup/
sudo wget

安装nginx所需依赖

nginx安装依赖命令
sudo yum -y install gcc zlib zlib-devel pcre-devel openssl openssl-devel

解压nginx

sudo tar -zxvf linux-nginx-1.10.2.tar.gz
cd nginx-1.10.2/
sudo ./configure /*编译解压
sudo make
sudo make install

编辑nginx

whereis nginx
cd /usr/local/nginx/conf/
sudo vim nginx.conf
添加一行：include vhost/.conf; /分号不能少

添加反向代理的配置文件

sudo mkdir vhost
cd vhost/
添加配置文件到vhost下

启动nginx

cd /usr/local/nginx/sbin
sudo ./nginx

查看nginx是否已经启动

ps -ef | grep nginx

重启nginx

/usr/nginx/sbin/nginx -s reload

安装mysql

检查是否安装了mysql

sudo rpm -qa | grep mysql-server

使用yum安装mysql

sudo yum -y install mysql-server

修改mysql字符集

sudo vim /etc/my.cnf

配置mysql的自启动

sudo chkconfig mysqld on
sudo chkconfig –list mysqld
sudo service mysqld restart

修改root密码

mysql -u root
select user,host,password from mysql.user
set password for root@localhost = password(‘自己设置的密码’); /*localhost,云服务器名，127.0.0.1分别设置一次
